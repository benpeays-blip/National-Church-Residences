<script>
/* Build swatches dynamically so tokens/labels stay in sync */
const blues = [
  {token:'--primary-900', label:'Brand Anchor / Header / Footer', use:'Header, primary nav, strongest accents'},
  {token:'--primary-800', label:'Strong Accent', use:'CTA hover, section dividers'},
  {token:'--primary-700', label:'Core Action', use:'Primary buttons, links'},
  {token:'--primary-600', label:'Secondary Accent', use:'Card headers, chips'},
  {token:'--primary-500', label:'Support Tone', use:'Table header, info sections'},
  {token:'--primary-400', label:'Light Neutral', use:'Cards/panels on tinted backgrounds'},
  {token:'--primary-300', label:'Very Light Neutral', use:'Alt sections, zebra rows'},
  {token:'--primary-100', label:'Base Background', use:'Page/body bg'}
];

const swWrap = document.getElementById('swatches');
blues.forEach(({token,label,use})=>{
  const div = document.createElement('div');
  div.className = 'sw';
  const top = document.createElement('div');
  top.className = 'sw-top';
  top.style.background = `var(${token})`;
  top.style.color = ['--primary-700','--primary-800','--primary-900'].includes(token) ? '#fff' : '#0f172a';
  top.innerHTML = `
    <div class="chip">Aa <span class="pill">${label}</span></div>
    <div class="mono copy" data-token="${token}">${token}</div>
  `;
  const foot = document.createElement('div');
  foot.className = 'sw-foot';
  foot.innerHTML = `<span class="mono">rgb ${getComputedStyle(document.documentElement).getPropertyValue(token).trim()}</span>
                    <span class="muted">${use}</span>`;
  div.appendChild(top); div.appendChild(foot);
  swWrap.appendChild(div);
});

// copy token name
document.addEventListener('click', e=>{
  const el = e.target.closest('.copy');
  if(!el) return;
  const txt = el.dataset.token;
  navigator.clipboard.writeText(`var(${txt})`);
  el.textContent = 'copied';
  setTimeout(()=>{el.textContent = txt}, 900);
});
</script>
