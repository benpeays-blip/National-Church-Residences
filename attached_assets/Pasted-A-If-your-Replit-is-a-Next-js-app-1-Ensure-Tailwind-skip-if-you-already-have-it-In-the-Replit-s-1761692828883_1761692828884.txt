A) If your Replit is a Next.js app
1) Ensure Tailwind (skip if you already have it)

In the Replit shell:

npm i -D tailwindcss postcss autoprefixer
npx tailwindcss init -p


Edit tailwind.config.js:

module.exports = { content: ["./src/**/*.{ts,tsx}"], theme: { extend: {} }, plugins: [] }


Add Tailwind to your global CSS (e.g., src/app/globals.css or styles/globals.css):

@tailwind base;
@tailwind components;
@tailwind utilities;

2) Add the registry file

Create src/data/integrations.json and paste your integration objects (start with a few; you can expand later). Example:

[
  {
    "slug": "salesforce-nonprofit-cloud",
    "name": "Salesforce Nonprofit Cloud",
    "category": "CRM",
    "logo": "/logos/salesforce.png",
    "description": "Enterprise donor CRM built on Salesforce.",
    "apiType": "OAuth",
    "modules": ["People","Gifts","Opportunities","Campaigns"],
    "features": ["Webhooks","Bi-directional sync"],
    "useCases": ["Constituent sync","Pipeline mirroring","Gift import"],
    "checklist": ["API credentials ready","Standardize designations","Define sync cadence"]
  }
]


Tip: put logo files in public/logos/ (e.g., public/logos/salesforce.png) and reference /logos/... for reliability.

3) Add the Gallery page

Create src/app/integrations/page.tsx (App Router) and paste the searchable gallery UI you want (from my prior message). It fetches /src/data/integrations.json. If your app uses Pages Router, put it in pages/integrations.tsx and convert to a client component.

4) Add the Detail page

Create src/app/integrations/[slug]/page.tsx and paste the detail + “Readiness Drawer” UI. (Pages Router: pages/integrations/[slug].tsx.)

5) Add a nav link (optional)

In src/app/layout.tsx (or your header component), add a link to /integrations.

6) Run
npm run dev


Open your Replit webview → visit /integrations. Click a card to open the detail page.

Troubleshooting

404 on JSON: ensure file path is exactly src/data/integrations.json and fetch path matches /src/data/integrations.json.

No styles: confirm Tailwind content points at ./src/**/*.{ts,tsx} and that your global CSS includes Tailwind directives.

B) If your Replit is a plain Node/Express app

You can still demo the gallery without converting to Next.js by serving one static SPA folder.

1) Add a minimal static SPA (Vite + React + Tailwind) inside /gallery

In shell:

npm create vite@latest gallery -- --template react-ts
cd gallery
npm i -D tailwindcss postcss autoprefixer
npx tailwindcss init -p


gallery/tailwind.config.js:

export default { content: ["./index.html","./src/**/*.{ts,tsx}"], theme: { extend: {} }, plugins: [] }


gallery/src/index.css:

@tailwind base;
@tailwind components;
@tailwind utilities;

2) Add the registry

Create gallery/public/integrations.json and paste your integration objects.

3) Build the SPA UI

Replace gallery/src/App.tsx with a simple gallery (search + filters + a detail drawer). (You can adapt the Next components I gave: change fetch path to /integrations.json, drop Next-specific bits.)

Quick skeleton:

// gallery/src/App.tsx
import { useEffect, useMemo, useState } from "react";
type Integration = { slug:string; name:string; category:string; logo:string; description:string; apiType:string; modules:string[]; features:string[]; useCases:string[]; checklist:string[] };

export default function App() {
  const [items, setItems] = useState<Integration[]>([]);
  const [q, setQ] = useState(""); const [cat, setCat] = useState("All"); const [open, setOpen] = useState<Integration|null>(null);
  useEffect(() => { fetch("/integrations.json").then(r=>r.json()).then(setItems); }, []);
  const cats = useMemo(()=>["All", ...Array.from(new Set(items.map(i=>i.category)))],[items]);
  const filtered = useMemo(()=>items.filter(i=>{
    const s = q.toLowerCase(); const byCat = cat==="All"||i.category===cat;
    const byQ = !q || i.name.toLowerCase().includes(s) || i.description.toLowerCase().includes(s) || i.modules.join(",").toLowerCase().includes(s);
    return byCat && byQ;
  }),[items,q,cat]);

  return (
    <main className="p-6 space-y-6">
      <h1 className="text-2xl font-semibold">Integrations Gallery</h1>
      <div className="flex flex-col md:flex-row gap-3">
        <input className="border rounded p-2 w-full md:w-1/2" placeholder="Search…" value={q} onChange={e=>setQ(e.target.value)}/>
        <select className="border rounded p-2 w-full md:w-60" value={cat} onChange={e=>setCat(e.target.value)}>{cats.map(c=><option key={c}>{c}</option>)}</select>
      </div>
      <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
        {filtered.map(i=>(
          <button key={i.slug} onClick={()=>setOpen(i)} className="text-left rounded-xl bg-white border shadow-sm hover:shadow-md p-4">
            <div className="flex items-center gap-3">
              <img src={i.logo} className="w-12 h-12 object-contain bg-white rounded"/>
              <div><div className="font-semibold">{i.name}</div><div className="text-xs text-gray-500">{i.category} • {i.apiType}</div></div>
            </div>
            <p className="text-sm mt-3 text-gray-700 line-clamp-3">{i.description}</p>
            <div className="flex gap-2 flex-wrap mt-3">{i.modules.slice(0,3).map(m=><span key={m} className="text-xs bg-gray-100 px-2 py-1 rounded">{m}</span>)}</div>
          </button>
        ))}
      </div>

      {open && (
        <div className="fixed inset-0 bg-black/30 flex">
          <div className="ml-auto w-full max-w-lg bg-white h-full p-6 shadow-xl">
            <div className="flex items-center justify-between">
              <h2 className="text-lg font-semibold">{open.name}</h2>
              <button onClick={()=>setOpen(null)} className="text-gray-500">✕</button>
            </div>
            <div className="mt-3 text-sm text-gray-600">{open.description}</div>
            <div className="mt-4 font-medium">Supported Modules</div>
            <div className="flex gap-2 flex-wrap mt-1">{open.modules.map(m=><span key={m} className="text-xs bg-gray-100 px-2 py-1 rounded">{m}</span>)}</div>
            <div className="mt-4 font-medium">Use Cases</div>
            <ul className="list-disc ml-5 text-sm">{open.useCases.map(u=><li key={u}>{u}</li>)}</ul>
            <div className="mt-4 font-medium">Readiness Checklist</div>
            <ul className="list-disc ml-5 text-sm">{open.checklist.map(c=><li key={c}>{c}</li>)}</ul>
          </div>
        </div>
      )}
    </main>
  );
}

4) Build the SPA
npm run build


(If Vite didn’t add scripts, add in gallery/package.json:

{ "scripts": { "dev": "vite", "build": "vite build", "preview": "vite preview --port 4173" } }


)

5) Serve the SPA from your existing Express app

At your project root (where your server.js / index.js lives):

Copy the built assets:

# from project root:
# ensure gallery build exists
cd gallery && npm run build && cd ..
# Vite outputs to gallery/dist — serve it under /integrations


In your Express server:

import path from "path";
import express from "express";
const app = express();

const galleryPath = path.join(process.cwd(), "gallery", "dist");
app.use("/integrations", express.static(galleryPath));
// SPA fallback:
app.get("/integrations/*", (req,res)=> res.sendFile(path.join(galleryPath,"index.html")));

// ...your existing routes...

app.listen(process.env.PORT || 3000);


Now navigate to /integrations in your live repl to see the gallery.
